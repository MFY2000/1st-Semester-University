float degart(float num){
	float pi = 3.14159265358979323846;

	int res1 = num * ( pi / 180 );
	
	return res1;
		
}

float caluMath(float num1 , float num2){
	
	float res =( sin(num1/2) * sin(num2/2) )+( cos(degart(num1)) * cos(degart(num2)) )*(sin(num1/2) , sin(num2/2));
	
	return res;	
}
